-----------------------------------------------------------------------------------------
POST /utilizadores ----------------------------------------------------------------------
- Regista um utilizador

request
    body = {username, nome, email, filiacao, curso, departamento, instituicao, password}

response
    success
        status_code = 201
        body = {
                   token,
                   utilizador: {_id, username, nome, email, filiacao, curso, departamento,
                                instituicao, admin, dataRegisto, ultimoAcesso}
               }
    error
        username/email already exists
            status_code = 409
        other
            status_code = 500
